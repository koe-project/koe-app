/** * GAS Backend API: Code.gs */ const SHEET_NAME = 'Answers'; const FOLDER_NAME = 'Koe_Voice_Records'; // スクリプトプロパティからAPIキーをセキュアに取得 const GEMINI_API_KEY = PropertiesService.getScriptProperties().getProperty("GEMINI_API_KEY"); /** * フロントエンドからのリクエストを受け取るエンドポイント */ function doPost(e) { const output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); try { const params = JSON.parse(e.postData.contents); const action = params.action; if (action === 'generateSteps') { const result = generateSteps(params.transcript, params.audioBase64); output.setContent(JSON.stringify({ success: true, data: result })); return output; } else if (action === 'saveAnswer') { const result = saveAnswer(params.data); output.setContent(JSON.stringify(result)); return output; } else { output.setContent(JSON.stringify({ success: false, error: 'Unknown action' })); return output; } } catch (err) { output.setContent(JSON.stringify({ success: false, error: err.toString() })); return output; } } function doGet(e) { return ContentService.createTextOutput("Koe API is running. MVP2 ready."); } function internal_insertMarkers(text) { if (!text) return ""; let t = "【（間）】" + text + "【（間）】"; const particles = ["は","が","を","に","で","の","と","も"]; const endings = ["かな","なぁ","ね","よ","ですよ","ました","です","ます","だった","だよね"]; const fillers = ["えー","あの","その","えっと","んー","うーん"]; const targets = [...fillers, ...endings, ...particles].sort((a,b)=>b.length-a.length); for (const w of targets) { const re = new RegExp(`${escapeRegExp(w)}(?!【（間）】)`, "g"); t = t.replace(re, `${w}【（間）】`); } return t; } function escapeRegExp(s){ return s.replace(/[.*+?^${}()|[\]\\]/g, "\\$&"); } /** * AIによる2ステップ文章整形 ＋ メタデータ自動抽出 */ function generateSteps(transcript, audioBase64) { if (!GEMINI_API_KEY) return JSON.stringify({ error: "APIキー未設定" }); const hasText = transcript && transcript.trim() !== ""; const instructionPrefix = hasText ? "内部的に挿入された呼吸マーカー【（間）】やスペースを元に、2つのステップで編集し、メタデータを抽出してください。" : "提供された音声を正確に文字起こしした上で、以下の2つのステップで編集し、メタデータを抽出してください。"; const markedText = hasText ? internal_insertMarkers(transcript) : "（音声から抽出してください）"; const prompt = ` 以下の指示に厳密に従って、作業対象のコンテンツを変換してください。 --- 作業指示 --- あなたは音声文字起こし原稿を整える編集者です。 ${instructionPrefix} Step1（句読点の補完）： ・話し言葉を維持しつつ、適切な読点「、」または句点「。」を補完してください。 Step2（読みやすく再構成）： ・Step1の結果を入力として、内容を理解しやすく再構成してください。 ・意味の重複や不要な言い淀みの削除、語順の整理を許可します。 ・ただし、話し言葉の温度感（〜かなぁ、〜だよね等）は残してください。 ★メタデータの抽出： 話の内容から、将来の年表作成のための情報を推測して付与してください。 ・tags: ["家族", "仕事", "趣味", "挑戦", "失敗", "学び", "友人", "その他"] から内容に合うものを1〜2つ選択。 ・era: "幼少期", "10代", "20代", "30代", "40代", "50代", "60代以上", "不明" から選択。（内容から推測できない場合は"不明"） ・emotion: "喜び", "悲しみ", "怒り", "驚き", "悔しさ", "感謝", "穏やか", "不明" から最も強く感じる感情を1つ選択。 ------------------ 作業対象： ${markedText} 必ず次のJSON形式のみで返してください。 { "step1": "Step1の結果", "step2": "Step2の結果", "tags": ["タグ1"], "era": "年代", "emotion": "感情" } `.trim(); const parts = [{ text: prompt }]; if (audioBase64) { parts.push({ inlineData: { mimeType: "audio/webm", data: audioBase64 } }); } const payload = { contents: [{ role: "user", parts: parts }], systemInstruction: { parts: [{ text: "あなたは正確な日本語変換作業者です。説明は禁止。必ずJSONのみ出力してください。" }] }, generationConfig: { responseMimeType: "application/json", temperature: 0.1 } }; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${GEMINI_API_KEY}`; const options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true }; try { const response = UrlFetchApp.fetch(url, options); const result = JSON.parse(response.getContentText()); if (result.candidates && result.candidates[0] && result.candidates[0].content) { return result.candidates[0].content.parts[0].text; } throw new Error("AI生成に失敗しました"); } catch (e) { return JSON.stringify({ error: e.toString() }); } } /** * データの保存 */ function saveAnswer(data) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { // カラムを追加 sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['questionId', 'weekNumber', 'audioUrl', 'transcriptText', 'step1', 'step2', 'tags', 'era', 'emotion', 'updatedAt']); } else { // 既存シートのヘッダー更新チェック (MVP1からMVP2への移行対応) const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (!headers.includes('tags')) { sheet.getRange(1, 7, 1, 4).setValues([['tags', 'era', 'emotion', 'updatedAt']]); } } let audioUrl = ""; if (data.audioBase64) { const folder = getOrCreateFolder(FOLDER_NAME); const blob = Utilities.newBlob(Utilities.base64Decode(data.audioBase64), 'audio/webm', `voice_${Date.now()}.webm`); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); audioUrl = file.getUrl(); } // タグ配列を文字列化 const tagsStr = data.tags && Array.isArray(data.tags) ? data.tags.join(",") : ""; sheet.appendRow([ data.questionId || "", data.weekNumber || "", audioUrl, data.transcriptText || "(音声から直接解析)", data.step1 || "", data.step2 || "", tagsStr, data.era || "", data.emotion || "", new Date().toISOString() ]); return { success: true }; } catch (e) { console.error("Save Error: " + e.toString()); return { success: false, error: e.toString() }; } } function getOrCreateFolder(name) { const folders = DriveApp.getFoldersByName(name); return folders.hasNext() ? folders.next() : DriveApp.createFolder(name); }
